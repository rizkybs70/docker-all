dnscrypt-proxy.toml ada di folder

docker run -d \
    --name dnscrypt \
	-e TZ=Asia/Jakarta \
    -p 5353:5353/tcp \
    -p 5353:5353/udp \
    -v /root/dnscrypt/dnscrypt-proxy.toml:/etc/dnscrypt-proxy/dnscrypt-proxy.toml \
	--restart unless-stopped \
    gists/dnscrypt-proxy:latest
